/*

Напишите функцию, которая


* называется IsPalindrom

* возвращает bool

* принимает параметр типа string

и возвращает, является ли

переданная строка палиндромом


Палиндром - это слово или фраза, которые одинаково читаются слева направо и справа налево.

*/


#include <iostream>

#include <string>


using namespace std;


bool IsPalindrom(string s)

{

string temp;

for(int i = s.size() - 1; i >= 0; --i)

{

temp += s[i];

}


if(s == temp) return true;

else return false;

}


int main()

{

string s;

cin >> s;


cout << IsPalindrom(s);


getchar();

return 0;

}
