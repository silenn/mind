#include <iostream>

#include <algorithm>

#include <vector>


using namespace std;


int main()

{

unsigned int n;

vector<int> v;

cin >> n;


while(n)

{

int num;

cin >> num;

v.push_back(num);

--n;

}


sort(v.begin(), v.end(), [](int i,int j) { return (abs(i)<abs(j)); });


for(const auto& i : v)

{

cout << i << " ";

}


getchar();

return 0;

}
