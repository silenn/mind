//main.c


/* USER CODE BEGIN 0 */

void stupidDelay(void);

/* USER CODE END 0 */


/* USER CODE BEGIN 2 */

HAL_TIM_PWM_Start(&htim4, TIM_CHANNEL_1);

HAL_TIM_PWM_Start(&htim4, TIM_CHANNEL_2);

HAL_TIM_PWM_Start(&htim4, TIM_CHANNEL_3);

HAL_TIM_PWM_Start(&htim4, TIM_CHANNEL_4);

/* USER CODE END 2 */


/* USER CODE BEGIN WHILE */

while (1)

{

/* USER CODE END WHILE */


/* USER CODE BEGIN 3 */

uint16_t i = 0;

for(; i < 65535; ++i)

{

TIM4->CCR1 = i;

stupidDelay();

}

for(; i > 0; --i)

{

TIM4->CCR1 = i;

stupidDelay();

}

}

/* USER CODE END 3 */


/* USER CODE BEGIN 4 */

void stupidDelay(void)

{

for(uint8_t delay = 0; delay < 255; ++delay) {}

}

/* USER CODE END 4 */
