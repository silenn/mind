#define MENU_MAIN 0

#define MENU_STATS 1

#define MENU_CONF 2

#define MENU_LOGS 3

#define MENU_UPDATE 4

#define MENU_REBOOT 5

#define MENU_CONF_SYS 21

#define MENU_CONF_IP 22

#define MENU_CONF_TIME 23


#define MENU_NOCHOISE 255


void menuStdItem(void);

void menuTipItem(void);

void menuMainList(void);

void menuConfigSysList(void);

void menuConfigIpList(void);

void menuRebootList(void);


struct Menu

{

uint8_t items;

uint8_t level;

void (*show) (void);

};


int main(void)

{

/*

enum MenuPage {

MENU_MAIN,

MENU_STATS,

MENU_CONF,

MENU_LOGS,

MENU_UPDATE,

MENU_REBOOT

};


enum MenuPage menuPage = MENU_MAIN;

*/

uint8_t menuPage = MENU_MAIN;

uint8_t menuChoise;


struct Menu menuMain = {5, 0, menuMainList};

struct Menu menuStatus = {2, 10, menuStatusList};

struct Menu menuConfig = {3, 20, menuConfigList};

struct Menu menuLogs = {1, 30, menuLogsList};

struct Menu menuUpdate = {1, 40, menuUpdateList};

struct Menu menuReboot = {2, 50, menuRebootList};

struct Menu menuConfigSys = {1, 120, menuConfigSysList};

struct Menu menuConfigIp = {6, 120, menuConfigIpList};

struct Menu menuConfigTime = {3, 120, menuConfigTimeList};


struct Menu* currentMenu;

currentMenu = &menuMain;


while(1)

{

//menuPage = currentMenu->level + menuChoise;


switch(menuPage)

{

case MENU_MAIN:

currentMenu = &menuMain;

break;

case MENU_STATS:

currentMenu = &menuStatus;

break;

case MENU_CONF:

currentMenu = &menuConfig;

break;

case MENU_LOGS:

currentMenu = &menuLogs;

break;

case MENU_UPDATE:

currentMenu = &menuUpdate;

break;

case MENU_REBOOT:

currentMenu = &menuReboot;

break;

case MENU_CONF_SYS:

currentMenu = &menuConfigSys;

break;

case MENU_CONF_IP:

currentMenu = &menuConfigIp;

break;

case MENU_CONF_TIME:

currentMenu = &menuConfigTime;

break;

default:

currentMenu = &menuMain;

break;

}


currentMenu->show();

if(currentMenu->level != MENU_MAIN) {

menuStdItem(); }

menuTipItem();


menuChoise = MENU_NOCHOISE;

while(menuChoise > currentMenu->items)

{

scanf("%u", &menuChoise);

}


if(menuChoise == 0 && currentMenu->level == 0)

{

printf("Logging out...\n");

getchar();

getchar();

return 0;

}

else

{

menuPage = currentMenu->level + menuChoise;

}


}

return 0;

}
