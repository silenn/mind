#include <iostream>

#include <vector>

#include <string>

#include <map>


using namespace std;


int main()

{

unsigned int q;

vector<string> bus_seq;

map<string, vector<string>> buses;

cin >> q;


while(q)

{

string command;

cin >> command;

if(command == "ALL_BUSES")

{


}

else

{

string busstop;

cin >> busstop;

if(command == "STOPS_FOR_BUS")

{

bool bus_exists = false;

unsigned int bus_seq_quantity = bus_seq.size();

for(unsigned int c=0; c<bus_seq_quantity; ++c)

{

if(bus_seq[c] == busstop)

{

bus_exists = true;

break;

}

}


if(bus_exists)

{

vector<string> stops = buses[busstop];

vector<string> search;

for(const auto& i : stops)

{

cout << "Stop " << i << ": ";


bool doYouFind = false;

for(unsigned int j=0; j<bus_seq_quantity; ++j)

{

if(bus_seq[j] != busstop) //number of bus

{

search = buses[bus_seq[j]]; //stops for bus bus_seq[j]

unsigned int quantity = search.size();

for(unsigned int k=0; k<quantity; ++k)

{

if(search[k] == i)

{

doYouFind = true;

cout << bus_seq[j] << " ";

break;

}

}

}


}

if(!doYouFind)

{

cout << "no interchange";

}

cout << "\n";

}

}

else

cout << "No bus\n";

/*for(unsigned int i=0; i<quantity; ++i)

{

cout << buses[busstop].second[i] << " ";

}*/

}

else if(command == "BUSES_FOR_STOP")

{

const auto quantity = bus_seq.size();

bool doYouFind = false;

for(unsigned int i=0; i<quantity; ++i)

{

for(const auto& j : buses[bus_seq[i]])

{

if(j == busstop)

{

doYouFind = true;

cout << bus_seq[i] << " ";

}

}

}

if(!doYouFind)

cout << "No stop";

cout << "\n";

}

else if(command == "NEW_BUS")

{

unsigned int stop_count;

string new_stop;

bus_seq.push_back(busstop);


cin >> stop_count;

while(stop_count)

{

cin >> new_stop;

buses[busstop].push_back(new_stop);

--stop_count;

}

}

}


--q;

}

system("pause");

return 0;

}
