Простые примеры работы с типами данных struct, vector, string и map:


#include <iostream>

#include <string>

#include <vector>

#include <map> //slovar'

using namespace std;


struct Person

{

string name;

string surname;

unsigned int age;

};


int main()

{

//<vector>

vector<int> nums = {1, 3, 5, 7};

cout << nums.size();


//<map>

map<string, int> name_to_value;

name_to_value["one"] = 1;

name_to_value["two"] = 2;

cout << "\ntwo is " << name_to_value["two"] << "\n";


//struct + vector

vector<Person> staff;

staff.push_back({"Ivan", "Ivanov", 25});

staff.push_back({"Petr", "Petrov", 32});

cout << staff[0].name;

return 0;

}
