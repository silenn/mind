#include <iostream>

#include <vector>

#include <set>


using namespace std;


void PrintSet(const set<string>& s)

{

cout << "Size: " << s.size() << "\n";

for(const auto& i : s)

cout << i << "\n";

}


int main()

{

set<string> famous_persons;

famous_persons.insert("Mask");

famous_persons.insert("Einstein");

famous_persons.insert("Winston");

famous_persons.erase("Winston");

PrintSet(famous_persons);


vector<string> v = {"Tesla", "Makswell", "Tesla"};

set<string> s(begin(v), end(v));

string x = "Hallo";

s.insert(x);

s.erase(x);

PrintSet(s);


getchar();

return false;

}
