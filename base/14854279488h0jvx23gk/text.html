/*

По введенному номеру дня недели вывести его название,

используя оператор множественного выбора.

Нумерация дней недели начинается с 1 – понедельник, 2 — вторник и т.д.

*/


#include <stdio.h>


int main()

{

unsigned char day;


printf("1. Monday\

\n2. Tuesday\

\n3. Wednesday\

\n4. Thursday\

\n5. Friday\

\n6. Saturday\

\n7. Sunday\n");


do

{

printf("Choose your day [1..7]: ");

scanf("%d", &day);

} while(day > 7 || day < 1);


switch(day)

{

case 1: printf("Monday\n"); break;

case 2: printf("Tuesday\n"); break;

case 3: printf("Wednesday\n"); break;

case 4: printf("Thursday\n"); break;

case 5: printf("Friday\n"); break;

case 6: printf("Saturday\n"); break;

case 7: printf("Sunday\n"); break;

}


getchar();

getchar();

return 0;

}
